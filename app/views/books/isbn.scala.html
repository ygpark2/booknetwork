ISBN Community

@bookTitle by @bookAuthor
Share My Post
@booksWithData.size reader posts are linked by ISBN @isbn.
@for(data <- booksWithData) { @defining(data.book) { book =>
@book.author.headOption.getOrElse('?')
@book.author @book.readingStatus.replace("_", " ").toLowerCase
@book.createdAt.format(java.time.format.DateTimeFormatter.ofPattern("MMM d"))
@book.reviewHeadline.getOrElse(book.title)
@book.title by @book.author
@book.rating.map { rating =>
@("★" * rating)@("☆" * (5 - rating))
}

@book.description.map(text => if (text.length > 220) text.take(220) + "..." else text).getOrElse("No reading note shared.")

@data.commentCount @data.repostCount @data.likeCount @data.bookmarkCount
} } }
